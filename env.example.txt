# Sitecore XM Cloud Configuration
# Copy this file to .env.local and fill in your values

# ========================================
# Authoring & Management GraphQL API
# (Required for fetching datasource field values)
# ========================================

# Your XM Cloud Authoring GraphQL endpoint
# Format: https://xmc-<your-org>-<environment>.sitecorecloud.io/sitecore/api/authoring/graphql/v1
NEXT_PUBLIC_AUTHORING_GRAPHQL_ENDPOINT=https://xmc-yourorg-environment.sitecorecloud.io/sitecore/api/authoring/graphql/v1

# Your XM Cloud API Key
# Get this from: XM Cloud Deploy > Project > Settings > Developer settings > API Keys
NEXT_PUBLIC_AUTHORING_API_KEY=your-api-key-here

# ========================================
# Experience Edge (Optional - for published content)
# ========================================

# Experience Edge Preview GraphQL Endpoint
NEXT_PUBLIC_EDGE_ENDPOINT=https://edge.sitecorecloud.io/api/graphql/v1

# Experience Edge API Key
NEXT_PUBLIC_EDGE_API_KEY=your-edge-api-key-here

# Site Name
NEXT_PUBLIC_SITE_NAME=your-site-name

# ========================================
# Azure OpenAI (Required for SEO Optimizer)
# ========================================

# Azure OpenAI Endpoint
# Format: https://<your-resource>.openai.azure.com/
NEXT_PUBLIC_AZURE_OPENAI_ENDPOINT=https://your-resource.openai.azure.com/

# Azure OpenAI API Key
# Get this from: Azure Portal > Your OpenAI Resource > Keys and Endpoint
NEXT_PUBLIC_AZURE_OPENAI_API_KEY=your-azure-openai-api-key-here

# Azure OpenAI Deployment Name
# The name of your deployed model (e.g., gpt-4, gpt-35-turbo)
NEXT_PUBLIC_AZURE_OPENAI_DEPLOYMENT=gpt-4

# ========================================
# Optional Settings
# ========================================

# Environment
NODE_ENV=development

# Enable debug logging
NEXT_PUBLIC_DEBUG=false
